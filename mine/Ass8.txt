using System;

public class Bao_Juhary_2F_AssignmentName_List_Processor
{
    public static void Main(string[] args)
    {
        string[] names = new string[5];
        int len = 0;
        for(int i = 0; i < 5; i++){
            Console.Write("Enter Name: ");
            names[i] = Console.ReadLine();
        }
        foreach (string name in names){
            Console.WriteLine(name);
            
            if (name.Length > 5){
                len++;
            }
        }
        
        Console.WriteLine("First Name: " + names[0]);
        Console.WriteLine("Last Name: " + names[4]);
        
        Console.WriteLine("names has more than 5 characters: " + len);
    }
}
